Programa interactivo para hacer zoom en im√°genes usando teclas: '+' para ampliar, '-' para reducir, 'o' para restaurar tama√±o original y 'x' para salir. 

def zoom():
    import cv2
    
    # Carga la imagen
    img_ori = cv2.imread("Spider.jpg")
    
    # Creando variable para el valor del zoom original
    zoom = 1.0

    while True:
        img = cv2.resize(img_ori, None, fx=zoom, fy=zoom)
        cv2.imshow("imagen nueva", img)
        tecla = cv2.waitKey(0)
        
        if tecla == ord('+'):
            zoom = zoom * 1.2
        elif tecla == ord('-'):
            zoom = zoom * 0.5
        elif tecla == ord('o'):
            zoom = 1.0
        elif tecla == ord('x'):
            break
    
    cv2.destroyAllWindows()

if __name__ == "__main__":
    zoom()
